        if(node->left){
            pii a=dfs(node->left);
            ans+=a.first;
            num+=a.second;

            pii a=dfs(node->right);
            ans+=a.first;
            num+=a.second;
        }
        if(ans/num==node->val){
            cnt++;
        }
        return make_pair(ans,num);
    }
};

                elif grid[i+1][j]=='(':
                    a|= dfs(i+1,j,s+1)
            return a
        if grid[0][0]!='(':
            return False
        return dfs(0,0,1)
