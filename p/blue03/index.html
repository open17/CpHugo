#define ios ios::sync_with_stdio(false);cin.tie(0);cout.tie(0);

typedef pair<int, int> pii;

set<pii> used;
int d[N];
int main() {
    ios
    int n,p,h,m,a,b;

    while(m--){
        cin>>a>>b;
        if(a>b)swap(a,b);
        if(used.count(make_pair(a,b)))continue;

        d[b]++;

        d[i]+=d[i-1];
        cout<<d[i]<<'\n';
    }
    return 0;
}
