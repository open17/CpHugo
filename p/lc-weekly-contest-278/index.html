class Solution:
    def subStrHash(self, s: str, power: int, modulo: int, k: int, hashValue: int) -> str:
        def f(p,m,start,end):
            hashv=0
            for i in range(start,end+1):
                hashv+=(ord(s[i])-ord('a')+1)*(p**(i-start))
            return hashv%m
        for i in range(k-1,len(s)):
            if f(power,modulo,i-k+1,i)==hashValue:
                return s[i-k+1:i+1]
        return s
